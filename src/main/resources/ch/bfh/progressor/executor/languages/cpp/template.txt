template<class T>
T machine_epsilon(T value) {

	union {
		int64_t i64;
		T d64;
	} u;

	u.d64 = value;
	u.i64++;

	return u.d64 - value;
}

template<class T>
bool hasMinimalDifference(T value1, T value2) {

	if (!isnormal(value1) || !isnormal(value2)) return value1 == value2;
	if (value1 == value2) return true;
	return abs(value1 - value2) <= machine_epsilon(value1);
}
